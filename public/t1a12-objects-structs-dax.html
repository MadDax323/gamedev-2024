More Examples of JavaScript Objects:

1. Nested Objects:
let myCar = {
    brand: "Ford",
    model: "Mustang",
    year: 2021,
    features: {
        color: "red",
        engine: "V8",
        seats: 4
    }
};
2. Array of Objects:
let people = [
    { firstname: "Alice", lastname: "Smith", age: 30 },
    { firstname: "Bob", lastname: "Johnson", age: 25 },
    { firstname: "Charlie", lastname: "Brown", age: 35 }
];
3. Object with Methods:
let myDog = {
    name: "Rex",
    breed: "Golden Retriever",
    bark: function() {
        return "Woof!";
    }
};
